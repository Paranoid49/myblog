### S-01：生产环境强制安全密钥

- 问题：`secret_key="change-me"` 默认值不阻止启动，生产环境下 session 可被伪造

- 预估工时：0.5h

```python
class Settings(BaseSettings):
    ...

    environment: str = "development"  # development / production

# app/main.py 启动阶段
```

- 开发模式保留 warning，生产模式直接阻止启动
- `.env.example` 同步新增 `ENVIRONMENT` 说明

### S-02：CSRF 保护

- 问题：Session/Cookie 认证无 CSRF token，存在跨站请求伪造风险
- 影响维度：安全性
- 预估工时：2h

整改方案（推荐方案 A：自定义头检查）：

后端：

```python
# app/core/deps.py 新增
def require_csrf_header(request: Request) -> None:
    """写接口要求携带自定义头，阻断浏览器默认表单提交的 CSRF 攻击"""
    if request.headers.get("X-Requested-With") != "XMLHttpRequest":
        raise HTTPException(
            status_code=403,
            detail=build_error_detail("csrf_rejected", 1005),
        )
```

```javascript
// frontend/src/shared/api/client.js
export async function apiRequest(path, options = {}) {
    const response = await fetch(`/api/v1${path}`, {
        credentials: 'same-origin',
        headers: {
            'X-Requested-With': 'XMLHttpRequest',
            ...options.headers,
        },
        ...options,
    });
    ...
}
```

- 登录接口当前是 Form 提交，需同步改为 fetch + 自定义头

### S-03：图片上传 Magic Bytes 校验

- 问题：仅校验客户端声明的 Content-Type，可伪造上传恶意文件
- 影响维度：安全性
- 预估工时：1h

整改方案：

```python
# app/routes/api_v1_media.py 新增
MAGIC_BYTES = {
    b'\xff\xd8\xff': "image/jpeg",
    b'\x89PNG\r\n\x1a\n': "image/png",
    b'RIFF': "image/webp",  # 需额外检查第 8-11 字节为 WEBP
    b'GIF87a': "image/gif",
    b'GIF89a': "image/gif",
}

def detect_image_type(content: bytes) -> str | None:
    """通过文件头 magic bytes 检测真实图片类型"""
    for magic, mime in MAGIC_BYTES.items():
        if content[:len(magic)] == magic:
            if magic == b'RIFF' and content[8:12] != b'WEBP':
                continue
            return mime
    return None
```

在 `upload_image_api` 中使用：

```python
detected_type = detect_image_type(content)
if detected_type is None or detected_type not in ALLOWED_IMAGE_TYPES:
    return error_response("unsupported_image_type", ...)
```

### S-04：安全响应头

- 问题：缺少 CSP、X-Frame-Options、X-Content-Type-Options 等安全头
- 影响维度：安全性
- 预估工时：1h

整改方案：

```python
# app/main.py 新增中间件
@app.middleware("http")
async def security_headers_middleware(request: Request, call_next):
    response = await call_next(request)
    response.headers["X-Content-Type-Options"] = "nosniff"
    response.headers["X-Frame-Options"] = "DENY"
    response.headers["X-XSS-Protection"] = "1; mode=block"
    response.headers["Referrer-Policy"] = "strict-origin-when-cross-origin"
    if request.url.path.startswith("/api/"):
        response.headers["Content-Security-Policy"] = "default-src 'none'"
    return response
```

### S-05：登录接口速率限制

- 问题：无防暴力破解措施（无速率限制、无失败计数）
- 影响维度：安全性
- 预估工时：2h

整改方案：

```python
# app/core/rate_limiter.py
import time
from collections import defaultdict

class InMemoryRateLimiter:
    """轻量级内存速率限制器，适合单进程部署"""

    def __init__(self, max_attempts: int = 5, window_seconds: int = 300):
        self.max_attempts = max_attempts
        self.window = window_seconds
        self._attempts: dict[str, list[float]] = defaultdict(list)

    def is_blocked(self, key: str) -> bool:
        now = time.monotonic()
        self._attempts[key] = [
            t for t in self._attempts[key] if now - t < self.window
        ]
        return len(self._attempts[key]) >= self.max_attempts

    def record(self, key: str) -> None:
        self._attempts[key].append(time.monotonic())

login_limiter = InMemoryRateLimiter(max_attempts=5, window_seconds=300)
```

在 `api_v1_auth.py` 中使用：

```python
@router.post("/login")
def api_login(request: Request, ...):
    client_ip = request.client.host if request.client else "unknown"

    if login_limiter.is_blocked(client_ip):
        return error_response("too_many_attempts", status.HTTP_429_TOO_MANY_REQUESTS, TOO_MANY_ATTEMPTS)

    user = authenticate_user(db, username, password)
    if not user:
        login_limiter.record(client_ip)
        return error_response("invalid_credentials", ...)

    request.session["user_id"] = user.id
    return ok_response(...)
```

- error_codes.py 新增 `TOO_MANY_ATTEMPTS = 1006`
- 保持轻量，不引入 Redis 等外部依赖

### D-05：结构化日志

- 问题：依赖默认 logging，无结构化日志、无日志级别配置
- 影响维度：部署与运维
- 预估工时：1h

```python
# app/core/logging_config.py
import logging
import sys


def setup_logging(environment: str = "development") -> None:
    """配置日志格式。生产环境使用 JSON 格式便于日志采集。"""
    if environment == "production":
        log_format = (
            '{"time":"%(asctime)s","level":"%(levelname)s",'
            '"logger":"%(name)s","message":"%(message)s"}'
        )
    else:
        log_format = "%(asctime)s %(levelname)-8s %(name)s  %(message)s"

    logging.basicConfig(
        level=logging.INFO,
        format=log_format,
        datefmt="%Y-%m-%d %H:%M:%S",
        stream=sys.stdout,
        force=True,
    )
```

在 `main.py` 顶部调用 `setup_logging(settings.environment)`。

步骤一：通用分页模型

```python
# app/schemas/pagination.py
from pydantic import BaseModel, Field


class PaginationParams(BaseModel):
    page: int = Field(default=1, ge=1)
    page_size: int = Field(default=20, ge=1, le=100)


def build_paginated_data(items: list, total: int, page: int, page_size: int) -> dict:
    return {
        "items": items,
        "total": total,
        "page": page,
        "page_size": page_size,
        "total_pages": (total + page_size - 1) // page_size,
    }
```

步骤二：service 层改造

```python
# app/services/post_service.py
from sqlalchemy import func

def list_published_posts(
    db: Session, page: int = 1, page_size: int = 20
) -> tuple[list[Post], int]:
    base = select(Post).where(Post.published_at.is_not(None))
    total = db.execute(select(func.count()).select_from(base.subquery())).scalar()
    posts = list(
        db.execute(
            base.order_by(Post.published_at.desc())
            .offset((page - 1) * page_size)
            .limit(page_size)
        ).scalars().all()
    )
    return posts, total
```

```python
@router.get("/posts", response_model=ApiResponse)
```

### P-02：初始化中间件缓存优化

- 问题：`_initialized_cache=True` 后仍创建 DB session
- 影响维度：性能
- 预估工时：0.5h

```python
# app/main.py
from app.services.setup_service import _initialized_cache, is_initialized

@app.middleware("http")
async def check_initialized_middleware(request: Request, call_next):
    path = request.url.path
    if (path.startswith("/static/") or path.startswith("/assets/")
        or path.startswith("/api/v1/setup") or path in SKIP_INIT_CHECK_PATHS):
        return await call_next(request)

    # 先检查内存缓存，命中则跳过 DB session 创建
    if _initialized_cache is True:
        return await call_next(request)

    with SessionLocal() as db:
        initialized = is_initialized(db)

    if not initialized:
        if path.startswith("/api/"):
            return JSONResponse(
                status_code=409,
                content=build_error_detail("site_not_initialized", SITE_NOT_INITIALIZED),
            )
        return RedirectResponse(url="/setup", status_code=302)

    return await call_next(request)
```

注意：需要将 `_initialized_cache` 改为模块级可导入变量，或提供 `is_cached_initialized()` 函数。

```python
    db: Session, slug: str, page: int = 1, page_size: int = 20
) -> tuple[Category | None, list[Post], int]:
    """在数据库层过滤已发布文章并分页"""
    category = db.execute(
        select(Category).where(Category.slug == slug)
    ).scalar_one_or_none()
    if not category:
        return None, [], 0

    base = select(Post).where(
        Post.category_id == category.id,
        Post.published_at.is_not(None),
    )
    total = db.execute(select(func.count()).select_from(base.subquery())).scalar()
    posts = list(
        db.execute(
            base.order_by(Post.published_at.desc())
            .offset((page - 1) * page_size)
            .limit(page_size)
        ).scalars().all()
    )
    return category, posts, total
```

- 路由层不再做 Python 列表过滤和排序

```python
def load_extensions(module_paths: str | None) -> tuple[list[str], list[str]]:
    """返回 (成功列表, 失败列表)"""
    if not module_paths:
        return [], []

    loaded: list[str] = []
    failed: list[str] = []
    for raw_path in module_paths.split(","):
        module_path = raw_path.strip()
        if not module_path:
            continue
        try:
            importlib.import_module(module_path)
            loaded.append(module_path)
            logger.info("扩展加载成功: %s", module_path)
        except Exception:
            logger.exception("扩展加载失败: %s", module_path)
            failed.append(module_path)
    return loaded, failed

# app/main.py
loaded, failed = load_extensions(settings.extension_modules)
if loaded:
    logger.info("已加载扩展: %s", ", ".join(loaded))
if failed:
    logger.warning("以下扩展加载失败: %s", ", ".join(failed))
```

```jsx
// frontend/src/shared/auth/AuthGuard.jsx
import { useEffect, useState } from 'react';
import { Navigate, useLocation } from 'react-router-dom';
import { apiRequest } from '../api/client';
import { hasStoredUserSnapshot, clearStoredUserSnapshot } from './session';

export default function AuthGuard({ children }) {
    const location = useLocation();
    const [checking, setChecking] = useState(true);
    const [valid, setValid] = useState(false);

    useEffect(() => {
        if (!hasStoredUserSnapshot()) {
            setChecking(false);
            return;
        }
        apiRequest('/taxonomy')
            .then(() => setValid(true))
            .catch(() => {
                clearStoredUserSnapshot();
                setValid(false);
            })
            .finally(() => setChecking(false));
    }, []);

    if (checking) {
        return <div className="notice muted">验证登录状态...</div>;
    }
    if (!valid) {
        return <Navigate to="/admin/login" replace state={{ from: location.pathname }} />;
    }
    return children;
}
```

### F-07：hook_bus 增加 unsubscribe

- 问题：一旦订阅无法取消，扩展无法干净卸载
- 影响维度：扩展性
- 预估工时：0.5h

```python
# app/core/hook_bus.py
def subscribe(self, event_name: str, handler: HookHandler) -> callable:
    """订阅事件，返回取消订阅函数"""
    self._handlers.setdefault(event_name, []).append(handler)

    def unsubscribe() -> None:
        try:
            self._handlers[event_name].remove(handler)
        except (KeyError, ValueError):
            pass

    return unsubscribe
```
